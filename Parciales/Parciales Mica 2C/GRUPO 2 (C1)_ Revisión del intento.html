Enunciado de la pregunta

Escriba un programa que permita calcular n valores de una función f(x) desde un valor inicial hasta un valor final, los grabe en un archivo y los dibuje por medio de  gnuplot. Utilice la f(x) que desee.

Obtendrá un puntaje extra :

Si escribe la función en un módulo.

Si escribe el script de gnuplot

Programa Principal


Program Parcial_PP

USE Parcial

implicit none


real(8) x , y , xi , xf , avance

integer n

Write(*,*) ' Ingrese el valor inicial de x' 

Read(*,*) xi

Write(*,*)' Ingrese el valor final de x'

Read(*,*) xf

Write(*,*)' Ingrese la cantidad de elementos'

Read(*,*) n


Open (unit=2, file='ejparcial.txt')

Do while (x<=xf)

 y=ejparcial(x)

 WRITE(*,'(2F15.6)', advance='No')x,y

 WRITE(2,'(2F15.6)', advance='No')x,y

 WRITE(*,*)' '

 WRITE(2,*)' '

 x=x+avance

END DO

Close (2)


CALL SYSTEM ("C:\gnuplot\bin\gnuplot.exe -persist script.p")


END PROGRAM


MODULO

module parcial


IMPLICIT NONE


CONTAINS

FUNCTION ejparcial(x)

real (8) x,ejparcial


ejparcial=x*x 


end function


end module


SCRIPT

# Ejemplo de script Gnuplot, para graficar los datos 

# que se encuentran en un archivo llamado "valores.dat"

# Este archivo se llama   ejemploScript.p

Explique con sus palabras que hace cada orden y que se obtiene como resultado en la variable f :

INTEGER f, n, i
f = 1
DO i=2, n
    f = i*f

f = 1 inicio f =1
