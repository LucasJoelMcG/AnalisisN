Enunciado de la pregunta

Escriba un programa que permita calcular n valores de la función f(x)= x3.e-0.5x desde un valor inicial X0 hasta un valor final Xf. Los valores de x y de f(x) deben ser almacenados en un archivo. El programa debe graficar la función f(x) por medio de gnuplot.

Obtendrá un puntaje extra:

Si escribe la función dentro de un módulo.

Si escribe el script de gnuplot.

Módulo:

MODULE parcial

IMPLICIT NONE

CONTAINS

FUNCTION ejparcial(x)

real (8), parameter :: e=2.718281828

real (8) x,ejparcial

ejparcial=x**(3)*e**(-0.5*x)

end function

END MODULE

Programa Principal:

PROGRAM principal_parcial

USE parcial

implicit none

real (8) x,y,xi,xf,avance

integer n

n=     !acá se debe dar un valor a la variable

xi=    !acá se debe dar un valor a la variable

xf=    !acá se debe dar un valor a la variable

x=xi

avance=(xf-xi)/(n-1.)

Open (unit=2, file='ejparcial.txt')

Do while (x<=xf)

 y=ejparcial(x)

 WRITE(*,'(2F15.6)', advance='No')x,y

 WRITE(2,'(2F15.6)', advance='No')x,y

 WRITE(*,*)' '

 WRITE(2,*)' '

 x=x+avance

END DO

Close (2)

CALL SYSTEM ("C:\gnuplot\bin\gnuplot.exe -persist script.p")


END PROGRAM

Script:

# Ejemplo de script Gnuplot, para graficar los datos 

# que se encuentran en un archivo llamado "valores.dat"

# Este archivo se llama   ejemploScript.p

set   autoscale                        # escala los ejes automaticamente

unset log                              # quita la escala logaritmica (si la hubiera)

unset label                            # quita los titulos anteriores

set xtic auto                          # establece automaticamente las divisiones del eje x

set ytic auto                          # establece automaticamente las divisiones del eje y

set yrange [-5:5]

set xrange [0.785398:2.356194]

set grid

set title "Funciones Parcial"

set xlabel "x"

set ylabel "y"


plot  "ejparcial.txt" using 1:2 title 'Funcion' with lines,\

      "ejparcial.txt" using 1:2 title 'Funcion' with points
